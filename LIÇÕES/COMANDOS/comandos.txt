use aula_03;




CREATE TABLE empresa (    
     id              INT auto_increment,    
     nome            VARCHAR(10),
     representante   varchar(05),
     primary key chave(id)
     ) auto_increment=100;


alter table empresa modify nome varchar(50), 
		    modify representante varchar (50);


desc empresa;

update empresa set funcionario = id * 10;
select * from empresa; 
select nome,
case when funcionario <= 1020 then 'Pequeno porte'
	 when funcionario < 1200 then 'Médio Porte'
	else 'Grande Porte' end Porte_Empresa
    From empresa;
 
SELECT nome, representante,
CASE 
    WHEN representante LIKE 'A%' THEN 'Começa com A' 
    ELSE 'Não Começa com A' 
END AS primeira_letra
FROM empresa;
